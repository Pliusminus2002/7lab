// assets/js/custom.js document.addEventListener("DOMContentLoaded", () => { const form = document.getElementById("contact-form"); if (!form) return; // jei netyčia skriptas įkeltas kitame puslapyje const resultBox = document.getElementById("form-result"); const avgBox = document.getElementById("avg-result"); const popup = document.getElementById("form-popup"); const popupText = popup ? popup.querySelector(".popup-text") : null; const popupClose = document.getElementById("popup-close"); // Pagalbinė funkcija – vidurkis function calculateAverage(q1, q2, q3) { return ( (q1 + q2 + q3) / 3 ); } form.addEventListener("submit", (e) => { e.preventDefault(); // NEleisti puslapiui persikrauti // Surenkam duomenis iš formos const data = { firstName: form.firstName.value.trim(), lastName: form.lastName.value.trim(), email: form.email.value.trim(), phone: form.phone.value.trim(), address: form.address.value.trim(), q1: Number(form.q1.value), q2: Number(form.q2.value), q3: Number(form.q3.value), }; // Paprastas patikrinimas – vertinimai turi būti 1–10 if ( data.q1 < 1 || data.q1 > 10 || data.q2 < 1 || data.q2 > 10 || data.q3 < 1 || data.q3 > 10 ) { alert("Visi vertinimai turi būti tarp 1 ir 10."); return; } // 4.b – išvedam objektą į naršyklės konsolę console.log("Kontaktų formos duomenys:", data); // 4.c – atvaizduojam tas pačias reikšmes formos apačioje if (resultBox) { resultBox.style.display = "block"; resultBox.innerHTML = ` Vardas: ${data.firstName}
Pavardė: ${data.lastName}
El. paštas: ${data.email}
Tel. numeris: ${data.phone || "-"}
Adresas: ${data.address || "-"}
Vertinimai (1–10): ${data.q1}, ${data.q2}, ${data.q3} `; } // 5 užduotis – apskaičiuojam vidurkį ir parodom formatu: // Vardas Pavardė: vidurkis (pvz., Vardenis Pavardenis: 9.8) const avg = calculateAverage(data.q1, data.q2, data.q3); if (avgBox) { avgBox.textContent = `${data.firstName} ${data.lastName}: vidurkis ${avg.toFixed(1)}`; } // 6 užduotis – parodom sėkmės pranešimą (pop-up) if (popup && popupText) { popupText.textContent = "Duomenys pateikti sėkmingai!"; popup.style.display = "flex"; } }); // Pop-up uždarymas if (popupClose && popup) { popupClose.addEventListener("click", () => { popup.style.display = "none"; }); } });
